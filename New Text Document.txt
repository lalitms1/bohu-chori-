    <h2>Bohu Chori (Stealing the Bride)</h2>
    
    <div id="game-container">
        <canvas id="gameCanvas" width="800" height="600"></canvas>
        
        <div id="ui-layer">
            <span id="lives-display">Lives: 3</span>
            <span id="status-display">Tag the Bohu!</span>
            <span id="timer-display">Time: 60</span>
            <button class="btn-green" onclick="openInstructions()">How to Play ⏸️</button>
        </div>

        <!-- Instructions Modal -->
        <div id="instructions-modal" class="overlay-card">
            <h2>How to Play</h2>
            <div class="instruction-text">
                <p><strong>🎯 Objective:</strong> You are the "Chora" (Blue). Run to the top of the village, tag the "Bohu" (Red), and escort her back to the Green safe zone at the bottom!</p>
                <p><strong>🕹️ Controls:</strong> Use <b>W A S D</b> or <b>Arrow Keys</b> on your keyboard to move around.</p>
                <p><strong>🏃 Enemies:</strong> Avoid the Defenders (Terracotta squares). If they catch you, you lose a life!</p>
                <p><strong>⏳ Time:</strong> You only have 60 seconds. Hurry!</p>
            </div>
            <button class="btn-green" style="margin-top: 15px;" onclick="closeInstructions()">Got it! Let's Play ▶️</button>
        </div>

        <!-- Game Over Modal -->
        <div id="game-over" class="overlay-card">
            <h1 id="end-title">Game Over!</h1>
            <p id="end-desc">The Defenders caught you.</p>
            <button onclick="resetGame()">Play Again 🔄</button>
        </div>

    <script>
        // ==========================================
        // 1. GAME SETUP & CANVAS INITIALIZATION
        // ==========================================
        const canvas = document.getElementById("gameCanvas");
        const ctx = canvas.getContext("2d");

        // Game State Variables
        let isPlaying = true;
        let isPaused = true; // Game starts paused so instructions can be read
        let lives = 3;
        let timeLeft = 60;

        // Input Tracking
        const keys = {
            ArrowUp: false, ArrowDown: false, ArrowLeft: false, ArrowRight: false,
            w: false, a: false, s: false, d: false
        };

        window.addEventListener("keydown", (e) => { if(keys.hasOwnProperty(e.key)) keys[e.key] = true; });
        window.addEventListener("keyup", (e) => { if(keys.hasOwnProperty(e.key)) keys[e.key] = false; });

        // ==========================================
        // 2. UI / OVERLAY MANAGEMENT
        // ==========================================

        function openInstructions() {
            isPaused = true; // Pause the game loop and timer
            document.getElementById("instructions-modal").style.display = "block";
        }

        function closeInstructions() {
            document.getElementById("instructions-modal").style.display = "none";
            isPaused = false; // Resume the game
            gameLoop(); // Restart the animation frame
        }

        // Open instructions automatically on first load
        window.onload = () => {
            openInstructions();
        };

        // ==========================================
        // 3. ENTITY DEFINITIONS
        // ==========================================

        class Chora {
            constructor() {
                this.width = 40;
                this.height = 40;
                this.resetPosition();
                this.speed = 4;
                this.color = "#0077b6"; 
                this.hasBohu = false;
            }

            resetPosition() {
                this.x = canvas.width / 2 - 20;
                this.y = canvas.height - 60; 
            }

            update() {
                if ((keys.ArrowUp || keys.w) && this.y > 0) this.y -= this.speed;
                if ((keys.ArrowDown || keys.s) && this.y < canvas.height - this.height) this.y += this.speed;
                if ((keys.ArrowLeft || keys.a) && this.x > 0) this.x -= this.speed;
                if ((keys.ArrowRight || keys.d) && this.x < canvas.width - this.width) this.x += this.speed;
            }

            draw() {
                ctx.fillStyle = this.color;
                ctx.fillRect(this.x, this.y, this.width, this.height);
                
                if (this.hasBohu) {
                    ctx.fillStyle = "#d90429"; 
                    ctx.beginPath();
                    ctx.arc(this.x + this.width/2, this.y - 10, 8, 0, Math.PI * 2);
                    ctx.fill();
                }
            }
        }

        class Bohu {
            constructor() {
                this.width = 30;
                this.height = 30;
                this.x = canvas.width / 2 - 15;
                this.y = 50; 
                this.color = "#d90429"; 
                this.isRescued = false;
            }

            draw() {
                if (this.isRescued) return; 
                ctx.fillStyle = this.color;
                ctx.fillRect(this.x, this.y, this.width, this.height);
                
                ctx.fillStyle = "#ffb703"; 
                ctx.fillRect(this.x + 5, this.y + 5, 20, 20);
            }
        }

        class Defender {
            constructor(startX, startY) {
                this.width = 40;
                this.height = 40;
                this.x = startX;
                this.y = startY;
                this.speed = 2.5; 
                this.color = "#e76f51"; 
            }

            update(targetX, targetY) {
                if(!isPlaying || isPaused) return; // Stop moving if paused
                
                const dx = targetX - this.x;
                const dy = targetY - this.y;
                const dist = Math.hypot(dx, dy);

                if (dist > 0) {
                    this.x += (dx / dist) * this.speed;
                    this.y += (dy / dist) * this.speed;
                }
            }

            draw() {
                ctx.fillStyle = this.color;
                ctx.fillRect(this.x, this.y, this.width, this.height);
                
                ctx.fillStyle = "white";
                ctx.fillRect(this.x + 5, this.y + 5, 10, 10);
                ctx.fillRect(this.x + 25, this.y + 5, 10, 10);
            }
        }

        let player = new Chora();
        let bride = new Bohu();
        let enemies =[new Defender(canvas.width / 2 - 20, canvas.height / 2)];

        // ==========================================
        // 4. GAME LOGIC & UTILITIES
        // ==========================================

        function checkCollision(rect1, rect2) {
            return rect1.x < rect2.x + rect2.width &&
                   rect1.x + rect1.width > rect2.x &&
                   rect1.y < rect2.y + rect2.height &&
                   rect1.y + rect1.height > rect2.y;
        }

        function triggerGameOver(win) {
            isPlaying = false;
            document.getElementById("game-over").style.display = "block";
            const title = document.getElementById("end-title");
            const desc = document.getElementById("end-desc");

            if (win) {
                title.innerText = "Victory!";
                title.style.color = "#2a9d8f";
                desc.innerText = "You successfully rescued the Bohu!";
            } else {
                title.innerText = "Game Over!";
                title.style.color = "#e76f51";
                desc.innerText = lives <= 0 ? "You were caught by the Defenders!" : "Time ran out!";
            }
        }

        function resetGame() {
            lives = 3;
            timeLeft = 60;
            player.resetPosition();
            player.hasBohu = false;
            bride.isRescued = false;
            enemies =[new Defender(canvas.width / 2 - 20, canvas.height / 2)];
            document.getElementById("game-over").style.display = "none";
            document.getElementById("status-display").innerText = "Tag the Bohu!";
            updateUI();
            isPlaying = true;
            isPaused = false; // ensure unpaused
            gameLoop(); 
        }

        function updateUI() {
            document.getElementById("lives-display").innerText = `Lives: ${lives}`;
            document.getElementById("timer-display").innerText = `Time: ${timeLeft}`;
        }

        // Timer interval (won't count down if paused)
        setInterval(() => {
            if (isPlaying && !isPaused && timeLeft > 0) {
                timeLeft--;
                updateUI();
                if (timeLeft <= 0) triggerGameOver(false);
            }
        }, 1000);

        // ==========================================
        // 5. MAIN GAME LOOP
        // ==========================================

        function gameLoop() {
            // Stop rendering/updating if not playing OR if paused
            if (!isPlaying || isPaused) return;

            // --- UPDATE PHASE ---
            player.update();
            enemies.forEach(enemy => enemy.update(player.x, player.y));

            // Collision: Player vs Bohu
            if (!player.hasBohu && checkCollision(player, bride)) {
                player.hasBohu = true;
                bride.isRescued = true;
                document.getElementById("status-display").innerText = "Run to the Start!";
            }

            // Collision: Player vs Defender
            enemies.forEach(enemy => {
                if (checkCollision(player, enemy)) {
                    lives--;
                    updateUI();
                    if (lives <= 0) {
                        triggerGameOver(false);
                    } else {
                        player.resetPosition();
                        enemy.x = canvas.width / 2 - 20;
                        enemy.y = canvas.height / 2;
                    }
                }
            });

            // Win Condition
            if (player.hasBohu && player.y > canvas.height - 80) {
                triggerGameOver(true);
            }

            // --- DRAW PHASE ---
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Safe Zone (Green)
            ctx.fillStyle = "rgba(42, 157, 143, 0.4)";
            ctx.fillRect(0, canvas.height - 80, canvas.width, 80);
            
            // Target Zone (Yellow)
            ctx.fillStyle = "rgba(233, 196, 106, 0.5)"; 
            ctx.fillRect(0, 0, canvas.width, 100);

            bride.draw();
            player.draw();
            enemies.forEach(enemy => enemy.draw());

            requestAnimationFrame(gameLoop);
        }

        // Initial draw before loop starts
        ctx.fillStyle = "rgba(42, 157, 143, 0.4)";
        ctx.fillRect(0, canvas.height - 80, canvas.width, 80);
        ctx.fillStyle = "rgba(233, 196, 106, 0.5)"; 
        ctx.fillRect(0, 0, canvas.width, 100);
        bride.draw();
        player.draw();
        enemies.forEach(enemy => enemy.draw());

    </script>
